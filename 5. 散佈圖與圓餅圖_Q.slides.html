散佈圖¶

import matplotlib.pyplot as plt  #引用

plt.scatter(x, y, s, c, alpha, marker)   #繪圖

參數說明：
    x, y : 在(x, y)的位置繪圖
    s : 點的大小
    c : 點的顏色
    alpha : 透明度
    marker : 點的樣式(https://matplotlib.org/stable/api/markers_api.html)

## 範例
import matplotlib.pyplot as plt 
import random 
math = [random.randint(60, 100) for i in range(50)] + [random.randint(10, 60) for i in range(50)]
science = [random.randint(60, 100) for i in range(50)] + [random.randint(10, 60) for i in range(50)]
english = [random.randint(10, 60) for i in range(50)] + [random.randint(60, 100) for i in range(50)]

plt.scatter(math, science, label="science")
plt.scatter(math, english, label="english")
plt.xticks(list(range(0, 101, 10)))
plt.yticks(list(range(0, 101, 10)))
plt.xlabel("Math Score")
plt.ylabel("Compare Score")
plt.title("Multi Data Scatter Example")
plt.legend()
plt.show()

圓餅圖¶

import matplotlib.pyplot as plt  #引用

plt.pie(data, labels, labeldistance, colors, autopct="%.1f%%", explode, startangle)   #繪圖

# data: 資料
# labels: 資料對應的標籤
# labeldistance: 標籤距離
# color: 資料對應的顏色
# autopct: 資料百分比
# explode: 區塊分離
# startangle: 起始角度

環狀圓餅圖¶

  • 環狀圓餅圖的概念即是在同一張畫布上畫兩次圓餅圖
    • 為了怕外部的百分比壓到內部的圓, 需設定pctdistance
    • 內部的圓需設定半徑radius否則會壓到外部的圓
In [116]:
## 範例 - 環狀圓餅圖
import pandas as pd
import matplotlib.pyplot as plt
import numpy as np

plt.figure(figsize=(5, 5))
data = np.random.randint(1, 10, 5)
print(data)
plt.pie(data,labels=range(5), autopct="%d%%", pctdistance=0.8)  #第一個繪製的圓餅圖為外圓, 需設pctdistance
plt.pie([0, 1], radius=0.6, colors='w')  #第二個繪製的圓餅圖需設radius
plt.title("Random Value")
plt.show()
